BNF

Entity IDs

entityURI

=

gatewayUrl "/" dataspaceKey [ "/" entityKey ]

dataspaceKey

=

entityKey

entityKey

=

1*( ALPHA / DIGIT / "." / "-" / "_" / "$" )


Query URLs

queryURL

=

entityURL  [ "/" selectorPath ]  [ queryString ]

queryString

=

"?" queryParam *( "&" queryParam )

queryParam

=

fieldsParam / sortParam / formatParam / langParam / maxArrayParam

 

fieldsParam

=

"fields=" selectorPath *( "," selectorPath )

sortParam

=

"sort=" selectorPath *( "," selectorPath )

formatParam

=

"format=" ( "json" / "xml" )

langParam

=

"lang=" iso639code

maxArrayParam

=

"maxArray=" 1*DIGIT

 

selectorPath

=

selectorSegment *( "/" selectorSegment )

selectorSegment

=

propertyName *( ";" patternExpr )

patternExpr

=

propertyName "=" value

propertyName

=

1*ALPHA

value

=

percent-encoded String value.  See conversion notes for appropriate value formats.

 

Fields Parameter

fieldsQueryParam

=

"fields=" plist

plist

=

"" | ( prule * ( "," prule ) )

bracketedList

=

"[" plist "]"

prule

=

( pdef | ruleName | ruleDef )

 

pdef

=

propertyName

[ ":" ( bracketedList | ruleName | derivation ) ]

[ "*" pageSize ]

[ ( "!" | "orderby" ) sortDef * ( ";" sortDef ) ]

propertyName

=

1*ALPHA

pageSize

=

int

sortDef

=

( "-" ) propertyName

 

ruleName

=

"@" propertyName

ruleDef

=

rulename "=" ( bracketedList | derivation  )
